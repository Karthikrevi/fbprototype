PART 2 — Token generation helper:

Add helper function in main.py:

import secrets
from datetime import datetime, timedelta

def generate_approval_token(
    user_type, user_id, user_email):

    token = secrets.token_urlsafe(32)
    expires_at = (
        datetime.now() +
        timedelta(hours=24)
    ).isoformat()

    conn = sqlite3.connect('erp.db')
    c = conn.cursor()
    c.execute("""
        INSERT INTO approval_tokens
        (token, user_type, user_id,
         user_email, expires_at)
        VALUES (?, ?, ?, ?, ?)
    """, (token, user_type,
          user_id, user_email,
          expires_at))
    conn.commit()
    conn.close()
    return token
