PART 3 — Update /pet-friendly route:

Replace the entire pet_friendly route
with this clean version:

@app.route('/pet-friendly')
def pet_friendly():
    if 'user_email' not in session:
        return redirect(url_for('login'))

    import json
    try:
        with open(
        'static/data/travel_affiliates.json'
        ) as f:
            affiliates = json.load(f)
    except:
        affiliates = {
            'hotels': [],
            'restaurants': [],
            'activities': []
        }

    conn = sqlite3.connect('erp.db')
    c = conn.cursor()
    c.execute("""
        SELECT * FROM pet_friendly_venues
        WHERE submission_status IN
        ('approved', 'auto_approved')
        AND is_active = 1
        ORDER BY created_at DESC
        LIMIT 20
    """)
    community_venues = c.fetchall()
    conn.close()

    return render_template(
        'pet_friendly.html',
        affiliates=affiliates,
        community_venues=community_venues)
