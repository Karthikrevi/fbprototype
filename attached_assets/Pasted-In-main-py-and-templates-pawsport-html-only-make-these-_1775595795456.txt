FIX 1 — Age calculation:

In the pet_passport route in main.py:
Calculate age dynamically from
the pet's birthday field.

from datetime import date

def calculate_pet_age(birthday_str):
    if not birthday_str:
        return 'Unknown'
    try:
        birthday = date.fromisoformat(
            birthday_str)
        today = date.today()
        years = today.year - birthday.year
        months = today.month - birthday.month
        if today.day < birthday.day:
            months -= 1
        if months < 0:
            years -= 1
            months += 12
        if years == 0:
            return f'{months} months old'
        elif months == 0:
            return f'{years} years old'
        else:
            return (f'{years} years '
                    f'{months} months old')
    except:
        return 'Unknown'

Pass calculated_age to pawsport.html
Replace "Age calculated" text with
{{ calculated_age }}
