Replace existing dropdown JavaScript
with this exact script:

<script>
document.addEventListener(
  'DOMContentLoaded', function() {
  
  const speciesSelect = 
    document.getElementById('species');
  const breedSelect = 
    document.getElementById('breed');
  const bloodSelect = 
    document.getElementById('blood_type');

  let breedsData = {};
  let bloodData = {};

  async function loadData() {
    try {
      const [breedsRes, bloodRes] = 
        await Promise.all([
          fetch('/static/data/breeds.json'),
          fetch('/static/data/blood_types.json')
        ]);
      breedsData = await breedsRes.json();
      bloodData = await bloodRes.json();
      
      if (speciesSelect.value) {
        updateBreeds(speciesSelect.value);
        updateBloodTypes(speciesSelect.value);
      }
    } catch (error) {
      console.error(
        'Error loading data:', error);
    }
  }

  function updateBreeds(species) {
    const breeds = breedsData[species] 
      || ['Mixed Breed', 'Other'];
    const currentBreed = 
      breedSelect.dataset.current || '';
    
    breedSelect.innerHTML = 
      '<option value="">Select breed</option>';
    breeds.forEach(breed => {
      const opt = 
        document.createElement('option');
      opt.value = breed;
      opt.textContent = breed;
      if (breed === currentBreed) {
        opt.selected = true;
      }
      breedSelect.appendChild(opt);
    });
  }

  function updateBloodTypes(species) {
    const types = bloodData[species] 
      || bloodData['Other'] 
      || ['Unknown'];
    const currentBlood = 
      bloodSelect.dataset.current || '';
    
    bloodSelect.innerHTML = 
      '<option value="">Select blood type</option>';
    types.forEach(type => {
      const opt = 
        document.createElement('option');
      opt.value = type;
      opt.textContent = type;
      if (type === currentBlood) {
        opt.selected = true;
      }
      bloodSelect.appendChild(opt);
    });
  }

  speciesSelect.addEventListener(
    'change', function() {
      updateBreeds(this.value);
      updateBloodTypes(this.value);
  });

  loadData();
});
</script>

today = date.today().isoformat()

c.execute("""
    SELECT b.*, v.business_name as vendor_name
    FROM bookings b
    LEFT JOIN vendors v ON b.vendor_id = v.id
    WHERE b.user_email = ?
    AND LOWER(b.pet_name) = LOWER(?)
    AND b.appointment_date >= ?
    AND b.status NOT IN 
      ('cancelled', 'completed')
